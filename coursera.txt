name = input("Enter file:")
if len(name) < 1 : name = "mbox-short.txt"
handle = open(name)
lst=list()
ddd=dict()
for line in handle:
    line=line.rstrip()
    if not line.startswith('From:'):continue
    words=line.split()
    mail=words[1]
    #print(mail)
    ddd[mail]=ddd.get(mail,0)+1
    #print(mail,'new',ddd[mail])
#find most common word
largest=-1
theword=None
for k,v in ddd.items():
    #print(k,v)
    if v>largest:
        largest=v
        theword=k
print(theword,largest)
